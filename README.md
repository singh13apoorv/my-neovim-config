-- mkdir ~/.config
-- cd ~/.config
-- mkdir nvim
-- clone this repository in the nvim folder you just created 
    git@github.com:singh13apoorv/my-neovim-config.git
    --or
    https://github.com/singh13apoorv/my-neovim-config.git
-- nvim lua/apoorvsconfig/packer.lua

-- go back to home directory 
-- mkdir ~/.vim
